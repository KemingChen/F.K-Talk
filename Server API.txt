Server API

/addFriend
	SP, token, phone

	response Success
		MQTT{ // 要對 "加入方" 跟 "被加方" 都要送相對應的訊息
			action: "updateFriend",
			data: {name, phone, photo, mail, hasReadMsgId}
		}

	response Fail
		MQTT{
			action: "error",
			data: "Error Msg"
		}

/listFriend
	SP, token

	response Success
		MQTT{ 
			action: "listFriend",
			data: [{name, phone, photo, mail, hasReadMsgId}, {...}, ...]
		}

	response Fail
		MQTT{
			action: "error",
			data: "Error Msg"
		}

/sendMsg
	SP, token, phone, message

	response Success
		MQTT{ // 要對 "送方" 跟 "被送方" 都要送一樣的訊息
			action: "addMsg",
			data: {messageId, sender, receiver, message, timestamp}
		}

	response Fail
		MQTT{
			action: "error",
			data: "Error Msg"
		}

/readMsg
	SP, token, phone, hasReadMsgId

	response Success
		MQTT{
			action: "hasRead",
			data: {phone, hasReadMsgId}
		}

	response Fail
		MQTT{
			action: "error",
			data: "Error Msg"
		}

/listMsg
	SP, token, phone

	response Success
		MQTT{ // 只要給 SP 即可
			action: "listMsg",
			data: [{messageId, sender, receiver, message, timestamp}, ...]
		}

	response Fail
		MQTT{
			action: "error",
			data: "Error Msg"
		}

/delFriend
	SP, token, phone

	response Success
		MQTT{ // 要對 "刪除方" 跟 "被刪除方" 都要送相對應的訊息
			action: "deleteFriend",
			data: "phone"
		}

	response Fail
		MQTT{
			action: "error",
			data: "Error Msg"
		}
